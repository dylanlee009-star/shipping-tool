export default { async fetch(request) { // Handle CORS preflight if (request.method === 'OPTIONS') { return new Response(null, { headers: { 'Access-Control-Allow-Origin': 'https://dylanlee009-star.github.io', 'Access-Control-Allow-Headers': '*', 'Access-Control-Allow-Methods': 'GET, POST, OPTIONS' } }); } const url = new URL(request.url); // Token exchange — POST to / if (request.method === 'POST') { const body = await request.text(); const res = await fetch('https://api.ebay.com/identity/v1/oauth2/token', { method: 'POST', headers: { 'Content-Type': 'application/x-www-form-urlencoded', 'Authorization': request.headers.get('Authorization') }, body }); const data = await res.text(); return new Response(data, { headers: { 'Access-Control-Allow-Origin': 'https://dylanlee009-star.github.io', 'Content-Type': 'application/json' } }); } // Orders fetch — GET to /orders if (request.method === 'GET' && url.pathname.startsWith('/orders')) { const filter = url.search || ''; const res = await fetch(`https://api.ebay.com/sell/fulfillment/v1/order${filter}`, { headers: { 'Authorization': request.headers.get('Authorization'), 'Content-Type': 'application/json' } }); const data = await res.text(); return new Response(data, { status: res.status, headers: { 'Access-Control-Allow-Origin': 'https://dylanlee009-star.github.io', 'Content-Type': 'application/json' } }); } return new Response('Not found', { status: 404 }); } }
